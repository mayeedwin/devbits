# Quick data processing
from dataclasses import dataclass
from typing import List

@dataclass
class Post:
    title: str
    content: str
    tags: List[str]

def filter_posts(posts: List[Post], tag: str) -> List[Post]:
    return [post for post in posts if tag in post.tags]
